# The Death of the Mamluk Sultan
country_event = {
	id = ottoman_mamluk_war_events.1
	title = ottoman_mamluk_war_events.1.t
	desc = ottoman_mamluk_war_events.1.d
	picture = BATTLE_eventPicture

	fire_only_once = yes
	is_triggered_only = yes
	major = yes

	option = {
		name = ottoman_mamluk_war_events.1.a # Onwards to Cairo!
		tooltip = {
			MAM = {
				kill_ruler = yes
			}
			MAM = {
				release = SYR
			}
			SYR = { 
				change_religion = ROOT	
				change_government = monarchy 
				add_government_reform = iqta
			}
			create_subject = {
				subject_type = vassal
				subject = SYR
			}
			MAM = {
				add_country_modifier = {
					name = mam_turmoil
					duration = 5200
				}
			}
		}
		if = {
			limit = {
		        OR = {
		        	NOT = { tag = TUR }
		        	NOT = { has_dlc = "Domination" }
		        	AND = {
		        		has_global_flag = tur_has_taken_defeat_mamluks_mission
		        		TUR = { has_country_flag = tur_decided_to_go_eyalet }
		        	}
		        }
			}
			custom_tooltip = mam_5_years
		}
		hidden_effect = {
			add_accepted_culture = al_suryah_arabic
			MAM = {
				country_event = {
					id = ottoman_mamluk_war_events.2
				}
			}
		}
	}
}

				NOT = { owns = 361 } ## cairo ##

				}

		}
		NOT = {
			is_in_war = {
				casus_belli = cb_ottoman_invasion

		}
		OR = {
			MAM = { NOT = { owns = 361 } } ## cairo ##

				limit = { is_subject_of = MAM }

			}
			inherit = MAM
			every_owned_province = {
				limit = {
					OR = {
						region = egypt_region
						is_core = MAM
					}
				}
				add_province_modifier = {
					name = tur_mamluk_control
					duration = -1
					desc = until_80_absolutism
				}
			}
			set_country_flag = tur_annexed_mamluks
			egypt_region = {
				limit = {
