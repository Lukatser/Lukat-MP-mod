									region = italy_region
								}
							}

								OR = {
									region = south_german_region
									region = north_german_region
									region = france_region
									region = low_countries_region
									region = italy_region
								}
							}
						}
					}
					full_ai = yes
				}
			}
		}
	}
	
	mean_time_to_happen = {
		months = 1
	}

	}

		name = ae_historical_hre.100.a

			name = ae_hab_emperor
			duration = 55000
		}
	}
}


#Defenestration of Prague
country_event = {
	id = ae_historical_hre.200
	title = ae_historical_hre.200.t
	desc = ae_historical_hre.200.d
	picture = DEFENESTRATION_PRAGUE_eventPicture
	fire_only_once = yes
	trigger = {
		NOT = { ae_gamerule_hre_off = yes }
		
		is_emperor = yes
		hre_leagues_enabled = yes
		OR = {
			is_in_league_war = yes
			is_year = 1618
		}
		if = {
			limit = { exists = BOH }
			BOH = { is_subject_of = ROOT }
		}
		else = {
			any_owned_province = {
				OR = { 
					is_core = BOH
					culture = czech
				}
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}	
	
	immediate = {
	}
	
	
	option = { 
		name = ae_historical_hre.200.a 
		set_country_flag = ae_defenestration
		if = {
			limit = { exists = BOH }
			BOH = { grant_independence = yes }
			set_global_flag = ae_status_quo_boh_subject
		}
		else = {
			every_owned_province = {
				limit = { culture = czech }
				add_core = BOH
			}
			release = BOH
			set_global_flag = ae_status_quo_boh_owned
		}
		random_country = {
			limit = { is_league_enemy = ROOT }
			BOH = { join_league = PREV }
		}
		if = { 
			limit = { exists = PAL }
			if = {
				limit = {
					PAL = { is_league_friend = ROOT }
				}
				PAL = { leave_league = ROOT }
			}
			random_country = {
				limit = { is_league_enemy = ROOT }
				PAL = { join_league = PREV }
			}
			PAL = { create_union = BOH }
			BOH = { add_liberty_desire = -200 }
		}
		if = {
			limit = { is_in_league_war = no }
			random_country = {
				limit = { 
					is_league_enemy = ROOT 
					is_league_leader = yes
				}
				declare_war_with_cb = {
					who = ROOT
					casus_belli = cb_religious_league
				}	
			}
		}
	}
}

	}

		months = 12
	}	
	
	immediate = {
	}

		name = ae_historical_hre.200.a 
		if = {
			limit = { has_global_flag = ae_status_quo_boh_subject }
			BOH = { 
				grant_independence = yes
				change_religion = catholic
				ROOT = { 
					create_subject = {
						subject_type = integrated_personal_union
						subject = BOH
					}
				} 
			}
		}
		else_if = {
			limit = { has_global_flag = ae_status_quo_boh_owned }
			inherit = BOH
		}
	}
}

#Protestant North Germany
country_event = {
	id = ae_historical_reformation.100
	title = ae_historical_reformation.100.t
	desc = ae_historical_reformation.100.d
	picture = PROTESTANT_GREAT_REFORMER_eventPicture
	
	trigger = {
		NOT = { ae_gamerule_reformation_off = yes }
		OR = {
			ae_gamerule_reformation_on = yes 
			full_ai = yes
		}
		is_religion_enabled = protestant
		OR = { 
			capital_scope = { region = north_german_region }
			tag = PAL
		}
		NOT = { government = theocracy }
		OR = {
			religion = catholic
			AND = {
				NOT = { religion = protestant }
				OR = {
					tag = BRA
					tag = PRU
					tag = SAX
				}
			}
		}
	}
	
	mean_time_to_happen = {
		months = 24
	}	
	
	immediate = {
	}
	
	
	option = { 
		name = ae_historical_reformation.100.a 
		change_religion = protestant
		capital_scope = {
			limit = {
				NOT = { religion = protestant }
			}
			change_religion = protestant
			add_permanent_province_modifier = {
				name = "religious_zeal_at_conv"
				duration = 9000
			}
		}
		random_owned_province = {
			limit = {
				is_capital = no
				NOT = { religion = protestant }
			}
			change_religion = protestant
			add_permanent_province_modifier = {
				name = "religious_zeal_at_conv"
				duration = 9000
			}
		}
	}
}
