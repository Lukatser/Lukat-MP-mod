		months = 1
	}

	immediate = {
		hidden_effect = {
			every_country = {
				limit = { war_with = ROOT }

			}
		}
	}

	option = {
		name = EVHO_spain_and_portugal_events.3.a
		every_country = {
			limit = { is_supporting_independence_of = ROOT }
			create_alliance = ROOT
		}
		every_country = {
			limit = { is_rival = OVERLORD }
			add_opinion = {
			    who = ROOT
			    modifier = new_friends
			}
			hidden_effect = {
				remove_opinion = {
				    who = ROOT
				    modifier = no_longer_friends
				}
			}
		}
		OVERLORD = {
			add_opinion = {
			    who = ROOT
			    modifier = no_longer_friends
			}
			hidden_effect = {
				remove_opinion = {
				    who = ROOT
				    modifier = new_friends
				}
				every_province = {
					limit = { 
						country_or_non_sovereign_subject_holds = ROOT 
						OR = {
							continent = north_america
							continent = south_america
						}
					}
					remove_claim = PREV
					add_claim = PREV
				}
			}
		}
		ROOT = { grant_independence = yes }
		hidden_effect = {
			set_country_flag = gained_de_facto_independence
			capital_scope = {
				build_to_forcelimit = {
					infantry = 0.6
					cavalry = 0.2
					artillery = 0.2
				}
			}
		}
		ai_chance = {
			factor = 1
		}
	}
