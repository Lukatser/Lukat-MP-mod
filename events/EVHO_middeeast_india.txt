# Conquest of Burma
country_event = {
	id = EVHO_near_east_and_india.1
	title = EVHO_near_east_and_india.1.t
	desc = EVHO_near_east_and_india.1.desc
	picture = BATTLE_eventPicture
	
	trigger = {
		num_of_cities = 3
		capital_scope = { region = burma_region }
		NOT = { has_country_flag = burma_claim }
		OR = {
			is_year = 1525
			AND = {
				is_year = 1512
				NOT = { exists = SST }
			}
		}
		NOT = {
			any_country = {
				capital_scope = { region = burma_region }
				OR = {
					num_of_cities = 30
					num_of_provinces_owned_or_owned_by_non_sovereign_subjects_with = {
						region = burma_region
						value = 25
					}
				}
			}
		}
	}
	
	mean_time_to_happen = {
		months = 1
	}

	option = {
		name = EVHO_near_east_and_india.1.a
		set_country_flag = burma_claim
		if = {
			limit = { is_subject = yes overlord = { ai = yes } }
			grant_independence = yes
		}
		burma_region = {
			limit = {
				NOT = { is_core = ROOT }
				NOT = { is_permanent_claim = ROOT }
			}
			add_core = ROOT
		}
		591 = { add_core = ROOT }
		2404 = { add_core = ROOT }
		every_country = {
			limit = { 
				NOT = { tag = ROOT }
				capital_scope = { region = burma_region } 
				OR = {
					NOT = { is_subject_of = ROOT }
					is_subject_of_type = tributary_state
				}
			}
			ROOT = {
				add_casus_belli = {
					type = cb_total_war_of_annexation
					target = PREV
					months = 300
				}
			}
			hidden_effect = {
				add_trust = {
				    who = ROOT	
				    value = -400
				}
				add_opinion = {
				    who = ROOT
				    modifier = no_longer_friends
				}
			}
		}
	}
}
