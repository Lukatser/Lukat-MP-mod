country_event = { ## Railway modifier automatic removal event
	id = EVHO_events_technical.113
	picture = CONSTITUTION_eventPicture
	
	hidden = yes
	
	trigger = {
		any_province = {
			has_building = railroad
			has_province_modifier = evho_province_railroad_modifier
			OR = {
				devastation = 50
				AND = {
					NOT = { development = 20 }
					NOT = { any_neighbor_province = { has_province_modifier = evho_province_railroad_modifier } }
				}
			}
		}
	}
	
	mean_time_to_happen = {
		days = 2
	}
	
	option = {
		name = "colonial_border_fix.RHans2"
		every_province = {
			limit = {
				has_building = railroad
				has_province_modifier = evho_province_railroad_modifier
				OR = {
					devastation = 50
					AND = {
						NOT = { development = 20 }
						NOT = { any_neighbor_province = { has_province_modifier = evho_province_railroad_modifier } }
					}
				}
			}
			remove_province_modifier = evho_province_railroad_modifier
		}
	}
}

country_event = { ## PU will take the color of the overlord
	id = EVHO_events_technical.115
	picture = CONSTITUTION_eventPicture
	
	is_triggered_only = yes
	hidden = yes
	
	trigger = {
		evho_is_colour_shifting_subject_type_scripted_trigger = yes
		any_country = {
			is_subject = no
			NOT = { tag = ROOT }
			evho_is_overlord_or_above_of_scripted_trigger = { subject = ROOT }
		}
		OR = {
			NOT = { has_country_flag = EVHO_has_any_subject_color_at_all_flag }
			any_country = {
				NOT = { has_country_flag = EVHO_pushed_overlord_color_onto_@ROOT }
				is_subject = no
				NOT = { tag = ROOT }
				evho_is_overlord_or_above_of_scripted_trigger = { subject = ROOT }
			}
		}
	}

	option = {
		name = "colonial_border_fix.RHans2"
		if = {
			limit = { NOT = { has_country_flag = EVHO_has_any_subject_color_at_all_flag } }
			set_country_flag = EVHO_has_any_subject_color_at_all_flag
		}
		random_country = {
			limit = {
				is_subject = no
				NOT = { tag = ROOT }
				evho_is_overlord_or_above_of_scripted_trigger = { subject = ROOT }
			}
			set_country_flag = EVHO_pushed_overlord_color_onto_@ROOT
			ROOT = { change_country_color = { country = PREV } }
		}
	}
}

country_event = { ## Fixes PU color on independence or new overlord (this isn't perfect in terms of flags, but it hypothetically works)
	id = EVHO_events_technical.116
	picture = CONSTITUTION_eventPicture
	
	is_triggered_only = yes
	hidden = yes
	
	trigger = {
		has_country_flag = EVHO_has_any_subject_color_at_all_flag
		OR = {
			evho_is_colour_shifting_subject_type_scripted_trigger = no
			any_country = { 
				has_country_flag = EVHO_pushed_overlord_color_onto_@ROOT 
				OR = {
					has_country_flag = EVHO_has_any_subject_color_at_all_flag
					NOT = { evho_is_overlord_or_above_of_scripted_trigger = { subject = ROOT } }
				}
			}
		}
	}

	option = {
		name = "colonial_border_fix.RHans2"
		if = {
			limit = { evho_is_colour_shifting_subject_type_scripted_trigger = no }

			restore_country_color = yes
			clr_country_flag = EVHO_has_any_subject_color_at_all_flag
			every_country = { ## in case of colonies made during a personal union, so they don't remain the color of your old overlord ##
				limit = { is_colonial_nation_of = ROOT }
				change_country_color = { country = ROOT }
			}
		}
		if = {
			limit = {
				any_country = { 
					has_country_flag = EVHO_pushed_overlord_color_onto_@ROOT 
					OR = {
						has_country_flag = EVHO_has_any_subject_color_at_all_flag
						NOT = { evho_is_overlord_or_above_of_scripted_trigger = { subject = ROOT } }
					}
				}
			}
			every_country = {
				limit = {
					has_country_flag = EVHO_pushed_overlord_color_onto_@ROOT 
					OR = {
						has_country_flag = EVHO_has_any_subject_color_at_all_flag
						NOT = { evho_is_overlord_or_above_of_scripted_trigger = { subject = ROOT } }
					}
				}
				clr_country_flag = EVHO_pushed_overlord_color_onto_@ROOT 
			}
		}
	}
}
