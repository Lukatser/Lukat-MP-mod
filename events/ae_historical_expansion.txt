#Force Conquest of Granada
country_event = {
	id = ae_historical_expansion.100
	title = ae_historical_expansion.100.t
	desc = ae_historical_expansion.100.d
	picture = BORDER_TENSION_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		is_year = 1492 
		tag = CAS 
		is_at_war = no
		upper_andalucia_area = {
			NOT = { owned_by = ROOT }
		}
		NOT = { ae_gamerule_historical_expansion_off = yes }
		OR = {
			ae_gamerule_historical_expansion_on = yes 
			all_countries_including_self = {
				if = {
					limit = { 
						OR = {
							tag = PREV
							tag = CAS
							tag = ARA
							tag = NAV
							tag = POR
							tag = GRA
							any_owned_province = { 
								OR = {
									region = iberia_region
									region = maghreb_region 
								}
							}
							any_core_province = { 
								OR = {
									region = iberia_region
									region = maghreb_region 
								}
							}
						}
					}
					full_ai = yes
				}
			}
		}
	}
	
	mean_time_to_happen = {
		months = 12
	}
	
	option = {
		name = ae_historical_expansion.100.a
		set_global_flag = ae_castilegranada
		upper_andalucia_area = {
			cede_province = ROOT
			add_core = ROOT
		}
	}
}

#Force Division of Livonia
country_event = {
	id = ae_historical_expansion.103
	title = ae_historical_expansion.103.t
	desc = ae_historical_expansion.103.d
	picture = BORDER_TENSION_eventPicture
	fire_only_once = yes
	
	trigger = {
		owns = 36
		current_age = age_of_reformation
		OR = {
			is_year = 1580 
			AND = {
				is_at_war = yes
				NOT = { controls = 36 }
			}
		}
		any_country = { is_liv_partition_muscovy = yes }
		any_country = { is_liv_partition_sweden = yes }
		any_country = { is_liv_partition_denmark = yes }
		any_country = { is_liv_partition_plc = yes }
		
		
		NOT = { ae_gamerule_historical_expansion_off = yes }
		OR = {
			ae_gamerule_historical_expansion_on = yes 
			all_countries_including_self = {
				if = {
					limit = { 
						OR = {
							tag = PREV
							tag = SWE
							tag = RUS
							tag = MOS
							tag = NOV
							tag = DAN
							tag = PLC
							tag = LIT
							tag = POL
							tag = LIV
							tag = PSK
							is_liv_partition_muscovy = yes
							is_liv_partition_sweden = yes
							is_liv_partition_denmark = yes
							is_liv_partition_plc = yes 
							any_owned_province = { livonian_region = yes }
							any_core_province = { livonian_region = yes }
						}
					}
					full_ai = yes
				}
			}
		}
	}
	
	immediate = {
		random_country = {
			limit = { is_liv_partition_muscovy = yes }
			save_event_target_as = liv_partition_muscovy
		}
		random_country = {
			limit = { is_liv_partition_sweden = yes }
			save_event_target_as = liv_partition_sweden
		}
		random_country = {
			limit = { is_liv_partition_denmark = yes }
			save_event_target_as = liv_partition_denmark
		}
		random_country = {
			limit = { is_liv_partition_plc = yes }
			save_event_target_as = liv_partition_plc
		}
	}
	
	mean_time_to_happen = {
		months = 12
	}
	
	option = {
		name = ae_historical_expansion.103.a 
		set_global_flag = ae_livonia_divided
		every_province = {
			if = {
				limit = { estland_area = yes }
				cede_province = event_target:liv_partition_sweden
				add_core = event_target:liv_partition_sweden
			}
			if = {
				limit = { livland_area = yes }
				if = {
					limit = {
						event_target:liv_partition_plc = { army_size = event_target:liv_partition_sweden }
					}
					cede_province = event_target:liv_partition_plc
					add_core = event_target:liv_partition_plc
				}
				else = {
					cede_province = event_target:liv_partition_sweden
					add_core = event_target:liv_partition_sweden
				}
			}
			if = {
				limit = { kurland_area = yes }
				cede_province = KUR
				add_core = KUR
				event_target:liv_partition_plc = { create_march = KUR }
			}
			if = {
				limit = { province_id = 35 } #Ösel
				cede_province = event_target:liv_partition_denmark
				add_core = event_target:liv_partition_denmark
			}
			if = {
				limit = { province_id = 273 } #Latgalia
				cede_province = event_target:liv_partition_plc
				add_core = event_target:liv_partition_plc
			}
			if = {
				limit = { province_id = 38 } #Riga
				cede_province = RIG
				add_core = RIG
				if = {
					limit = {
						event_target:liv_partition_plc = { army_size = event_target:liv_partition_sweden }
					}
					event_target:liv_partition_plc = { create_march = RIG }
				}
				else = {
					event_target:liv_partition_sweden = { create_march = RIG }
				}
			}
		}
		every_country = {
			limit = { livonian_country = yes }
			every_owned_province = {
				if = {
					limit = { region = russia_region }
					cede_province = event_target:liv_partition_muscovy
				}
				else = { cede_province = event_target:liv_partition_plc }
			}
		}
	}
	
}
