# The Mishar Conversion

country_event = {
	id = EVHO_russia.2
	title = EVHO_russia.2.t
	desc = EVHO_russia.2.desc
	picture = COURT_eventPicture
	
	fire_only_once = yes
	
	major = yes

	trigger = {
		OR = {
			primary_culture = russian_culture
			primary_culture = russian_culture_new
		}
		OR = {
			tag = RUS
			tag = NOV
			tag = MOS
		}
		NOT = {
			any_province = {
				NOT = { country_or_non_sovereign_subject_holds = ROOT }
				culture = mishary
			}
			any_province = {
				country_or_non_sovereign_subject_holds = ROOT
				has_owner_religion = no
				culture = mishary
			}
		}
	}
	
	mean_time_to_happen = {
		months = 100
	}

	option = {
		name = EVHO_russia.2.a
		add_adm_power = -50
		add_mil_power = -50
		add_dip_power = -150
		remove_accepted_culture = mishary
		if = {
			limit = { primary_culture = russian_culture }
			every_province = {
				limit = { culture = mishary }
				change_culture = russian_culture
			}
		}
		else_if = {
			limit = { primary_culture = russian_culture_new }
			every_province = {
				limit = { culture = mishary }
				change_culture = russian_culture_new
			}
		}
		ai_chance = {
			factor = 90
		}
	}
}
