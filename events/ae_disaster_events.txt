country_event = {
	id = ae_decline_of_the_mughals.102
	title = ae_decline_of_the_mughals.102.t
	desc = ae_decline_of_the_mughals.102.d
	picture = 3_Rajput_Estate__eventPicture 
	
	is_triggered_only = yes
	fire_only_once = yes 
	
	trigger = { 
		any_owned_province = { culture = rajput }
		RJP = {	is_at_war = no }
	}
	
	
	option = { 
		name = ae_decline_of_the_mughals.102.a  
		add_stability = -1
		every_province = {
			limit = { culture = rajput }
			add_core = RJP
		}
		every_owned_province = {
			limit = { culture = rajput }
			area = { cede_province = RJP }
		}
		every_country = {
			limit = { primary_culture = rajput }
			RJP = { inherit = PREV }
		}
		RJP = {
			if = {
				limit = { 
					NOT = { army_size = ROOT }
				}
				capital_scope = {
					build_to_forcelimit = {
						infantry = 0.5
						cavalry = 0.3
						artillery = 0.1
					}
				}
			}
			declare_war_with_cb = {
				who = ROOT
				casus_belli = cb_annex
			}
		}
	}
}
