country_event = {
	id = EVHO_anglo_british_events.1
	title = EVHO_anglo_british_events.1.t
	desc = EVHO_anglo_british_events.1.d
	picture = GOOD_WITH_MONARCH_eventPicture

	major = yes
	fire_only_once = yes
	
	trigger = {
		tag = SCO
		ENG = { government = monarchy is_free_or_tributary_trigger = yes }
		SCO = { 
			government = monarchy 
			OR = {
				is_subject_of = ENG
				is_free_or_tributary_trigger = yes 
			}
		}
		NOT = { is_year = 1750 }
		valid_for_personal_unions_trigger = yes
		is_year = 1600
		SCO = {
			valid_for_personal_unions_trigger = yes
			OR = {
				ai = yes
				has_heir = no
				marriage_with = ENG
				has_opinion = { who = ENG value = 100 }
			}
		}
		NOT = {
			AND = {
				ENG = {
					OR = {
						has_ruler_flag = james_i
						has_heir_flag = james_i
					}
				}
				SCO = {
					OR = {
						has_ruler_flag = james_i
						has_heir_flag = james_i
					}
				}
			}
		}
	}
	
	mean_time_to_happen = {
		months = 500
		modifier = {
			factor = 0.1
			any_province = {
				region = british_isles_region
				owner = {
				is_great_power = yes
					capital_scope = {
						NOT = { region = british_isles_region }
					}
				}
			}
		}
		modifier = {
			factor = 0.1
			is_year = 1625
		}
		modifier = {
			factor = 0.05
			is_subject_of = ENG
		}
	}

	option = {
		name = EVHO_anglo_british_events.1.a
		trigger = { is_subject = no }
		ai_chance = { factor = 0 }
		ENG = {
			add_casus_belli = {	
				target = SCO
				type = cb_restore_personal_union
				months = 60	
			}
		}
	}
	option = {
		name = EVHO_anglo_british_events.1.b
		ai_chance = { factor = 100 }
		add_stability_or_adm_power = yes
		if = {
			limit = { is_subject = yes }
			hidden_effect = {
				grant_independence = yes
				set_country_flag = SCO_was_already_english_subject_flag
			}
		}
		ENG = { create_union = SCO }
		if = {
			limit = { has_country_flag = SCO_was_already_english_subject_flag }
			hidden_effect = { clr_country_flag = SCO_was_already_english_subject_flag }
			add_liberty_desire = -60
		}
	}
}

	id = EVHO_anglo_british_events.7

	desc = EVHO_anglo_british_events.7.d
	picture = TEA_PARTY_eventPicture
	
	major = yes
	fire_only_once = yes
	
	trigger = {
		is_colonial_nation = yes
		NOT = { exists = USA }
		is_year = 1773
		OR = {
			liberty_desire = 10
			OVERLORD = { ai = yes }
		}
		any_province = {
			has_port = yes
			is_in_capital_area = yes 
			OWNER = { tag = ROOT }
		}
		OR = {
			tag = TTC
			capital_scope = { 
				OR = {
					has_port = yes
					colonial_region = colonial_new_england
				}
				OR = {
					region = northeast_america_region
					region = southeast_america_region
				}
			}
		}
	}
	
	immediate = {
		hidden_effect = {
			if = {
				limit = { owns = 968 }
				968 = { save_event_target_as = boston_tea_party_province }
			}
			else_if = {
				limit = { capital_scope = { has_port = yes } }
				capital_scope = {
					save_event_target_as = boston_tea_party_province
				}
			}
			else = {
				random_owned_province = {
					limit = { 
						is_in_capital_area = yes 
						has_port = yes
					}
					save_event_target_as = boston_tea_party_province
				}
			}
		}
	}

	mean_time_to_happen = {
		months = 9
		modifier = {

			    who = ROOT
