#Force England-Scotland
country_event = {
	id = ae_historical_unions.100
	title = ae_historical_unions.100.t
	desc = ae_historical_unions.100.d
	picture = DIPLOMACY_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = ENG
		government = monarchy
		
		is_at_war = no
		SCO = { 
			exists = yes 
			dynasty = "Stuart"
			government = monarchy
			NOT = { junior_union_with = ENG }
			is_at_war = no
		} 
		is_subject = no
		OR = {
			is_year = 1600
			AND = {
				is_year = 1540
				NOT = {
					dynasty = "Lancaster"
					dynasty = "York"
					dynasty = "Tudor"			
				} 
			}
		}
		
		NOT = { ae_gamerule_unions_off = yes }
		OR = {
			ae_gamerule_unions_on = yes 
			all_countries_including_self = {
				if = {
					limit = { 
						OR = {
							tag = PREV
							tag = ENG
							tag = SCO
							any_owned_province = { 
								region = british_isles_region
							}
							any_core_province = { 
								region = british_isles_region
							}
						}
					}
					full_ai = yes
				}
			}
		}
	} 
	
	option = { 
		name = ae_historical_unions.100.a 
		set_global_flag = ae_engsco
		SCO = { exile_ruler_as = { name = ae_new_english_king } }
		ENG = { set_ruler = ae_new_english_king }
		create_union = SCO
		every_province = {
			limit = {
				OR = {
					area = lowlands_area 
					area = highlands_area
					area = kingdom_of_the_isles_area 
				}
				NOT = { province_id = 4365 }
			}
			cede_province = SCO 
		}
	}
}
