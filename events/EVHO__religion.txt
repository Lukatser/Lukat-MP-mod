#Irish Religious Difficulties
country_event = {
	id = ev_religion.1
	title = "ev_religion.1.t"
	desc = "ev_religion.1.d"
	picture = ANGRY_MOB_eventPicture

	mean_time_to_happen = {
		months = 3
	}

	trigger = {
		NOT = { has_country_flag = religion_difficult_ireland }
		NOT = { has_country_flag = evho_has_scripted_religious_seperatists_tolerated_flag }
		OR = {
			is_free_or_tributary_trigger = no
			NOT = { capital_scope = { evho_province_is_in_ireland = yes } }
		}
		any_province = {
			owned_by = ROOT
			NOT = { religion = ROOT }
			evho_province_is_in_ireland = yes
			NOT = { has_province_modifier = estate_dhimmi_guaranteed_autonomy_mod }
			NOT = { 
				AND = {
					religion = orthodox
					has_province_modifier = estate_burghers_guaranteed_autonomy_mod
				}
		 	}
			OR = {
				NOT = { culture = ROOT }
				ROOT = { is_free_or_tributary_trigger = no }
			}
		}
	}
	
	immediate = {
		set_country_flag = religion_difficult_ireland
	}

	option = {
		name = "ev_religion.1.a"
		every_owned_province = {
			limit = {
				NOT = { has_province_modifier = local_religious_difficulties }
				NOT = { religion = ROOT }
				evho_province_is_in_ireland = yes
				NOT = { has_province_modifier = estate_dhimmi_guaranteed_autonomy_mod }
				NOT = { 
					AND = {
						religion = orthodox
						has_province_modifier = estate_burghers_guaranteed_autonomy_mod
					}
			 	}
				OR = {
					NOT = { culture = ROOT }
					ROOT = { is_free_or_tributary_trigger = no }
				}
			}
			if = {
				limit = { has_province_modifier = estate_nobles_evho_placated_religious_seperatists_mod }
				remove_province_modifier = estate_nobles_evho_placated_religious_seperatists_mod
			}
			if = {
				limit = { has_province_modifier = estate_church_evho_right_to_inquisition_mod }
				remove_province_modifier = estate_church_evho_right_to_inquisition_mod
			}
			if = {
				limit = { has_province_flag = estate_nobles_evho_should_have_lesser_religious_difficulties }
				clr_province_flag = estate_nobles_evho_should_have_lesser_religious_difficulties
			}
			add_province_modifier = {
				name = local_religious_difficulties
				duration = -1
			}
		}
	}
}

country_event = {

	title = "ev_religion.1.t"
	desc = "ev_religion.1.d"
	picture = ANGRY_MOB_eventPicture

	mean_time_to_happen = {
		months = 1
	}

	hidden = yes

	trigger = {
		OR = {
			has_country_flag = religion_difficult_ireland
			has_country_flag = evho_has_scripted_religious_seperatists_tolerated_flag
		}
		OR = {
			is_free_or_tributary_trigger = no

			owned_by = ROOT

			OR = {
				NOT = { culture = ROOT }

		if = {

				 	}

						NOT = { culture = ROOT }
						ROOT = { is_free_or_tributary_trigger = no }
					}
				}
				remove_province_modifier = local_religious_difficulties
				clr_province_flag = estate_nobles_evho_should_have_lesser_religious_difficulties
				add_province_modifier = {
					name = estate_nobles_evho_placated_religious_seperatists_mod

				}
			}

			limit = { has_country_flag = evho_has_scripted_religious_seperatists_inquisition_flag }
			every_owned_province = {
				limit = {
					NOT = { has_province_modifier = estate_church_evho_right_to_inquisition_mod }
					NOT = { religion = ROOT }
					evho_province_is_in_ireland = yes
					NOT = { has_province_modifier = estate_dhimmi_guaranteed_autonomy_mod }
					NOT = { 
						AND = {
							religion = orthodox
							has_province_modifier = estate_burghers_guaranteed_autonomy_mod
						}
				 	}
					OR = {
						NOT = { culture = ROOT }
						ROOT = { is_free_or_tributary_trigger = no }
					}
				}
				remove_province_modifier = local_religious_difficulties
				clr_province_flag = estate_nobles_evho_should_have_lesser_religious_difficulties
				add_province_modifier = {
					name = estate_church_evho_right_to_inquisition_mod
					duration = -1
				}
			}
		}
		else = {
			every_owned_province = {
				limit = {
					NOT = { has_province_modifier = local_religious_difficulties }
					NOT = { religion = ROOT }
					evho_province_is_in_ireland = yes
					NOT = { has_province_modifier = estate_dhimmi_guaranteed_autonomy_mod }

					name = local_religious_difficulties
					duration = -1
				}
			}

	option = {

country_event = {
	id = ev_religion.3
	title = "ev_religion.1.t"
	desc = "ev_religion.1.d"
	picture = ANGRY_MOB_eventPicture

	mean_time_to_happen = {
		months = 1
	}

	hidden = yes

	trigger = {
		is_free_or_tributary_trigger = yes
		any_province = {
			owned_by = ROOT
			evho_province_is_in_ireland = yes
			OR = {
				has_province_modifier = local_religious_difficulties
				has_province_modifier = estate_nobles_evho_placated_religious_seperatists_mod
				has_province_modifier = estate_church_evho_right_to_inquisition_mod
			}
		}
		OR = {
			capital_scope = { evho_province_is_in_ireland = yes }
			any_province = {
				owned_by = ROOT
				evho_province_is_in_ireland = yes
				OR = {
					has_province_modifier = local_religious_difficulties
					has_province_modifier = estate_nobles_evho_placated_religious_seperatists_mod
					has_province_modifier = estate_church_evho_right_to_inquisition_mod
				}
				OR = {
					religion = ROOT
					has_province_modifier = estate_dhimmi_guaranteed_autonomy_mod
					AND = {
						religion = orthodox
						has_province_modifier = estate_burghers_guaranteed_autonomy_mod
					}
					AND = {
						culture = ROOT
						ROOT = { is_free_or_tributary_trigger = yes }
					}
					AND = {
						ROOT = { NOT = { has_country_flag = evho_has_scripted_religious_seperatists_tolerated_flag } }
						has_province_modifier = estate_nobles_evho_placated_religious_seperatists_mod
					}
					AND = {
						ROOT = { NOT = { has_country_flag = evho_has_scripted_religious_seperatists_inquisition_flag } }
						has_province_modifier = estate_church_evho_right_to_inquisition_mod
					}
				}
			}
		}
	}
	
	immediate = {
		if = {
			limit = { NOT = { has_country_flag = evho_has_scripted_religious_seperatists_tolerated_flag } }
			every_province = {
				limit = { has_province_modifier = estate_nobles_evho_placated_religious_seperatists_mod }
				remove_province_modifier = estate_nobles_evho_placated_religious_seperatists_mod
			}
		}
		if = {
			limit = { NOT = { has_country_flag = evho_has_scripted_religious_seperatists_inquisition_flag } }
			every_province = {
				limit = { has_province_modifier = estate_church_evho_right_to_inquisition_mod }
				remove_province_modifier = estate_church_evho_right_to_inquisition_mod
			}
		}
		if = {
			limit = { capital_scope = { evho_province_is_in_ireland = yes } }
			every_owned_province = {
				limit = {
					has_province_modifier = local_religious_difficulties
					evho_province_is_in_ireland = yes
				}
				remove_province_modifier = local_religious_difficulties
			}
			every_owned_province = {
				limit = {
					has_province_modifier = estate_nobles_evho_placated_religious_seperatists_mod
					evho_province_is_in_ireland = yes
				}
				remove_province_modifier = estate_nobles_evho_placated_religious_seperatists_mod
			}
			every_owned_province = {
				limit = {
					has_province_modifier = estate_church_evho_right_to_inquisition_mod
					evho_province_is_in_ireland = yes
				}
				remove_province_modifier = estate_church_evho_right_to_inquisition_mod
			}
		}
		else = {
			every_owned_province = {
				limit = {
					has_province_modifier = local_religious_difficulties
					evho_province_is_in_ireland = yes
					OR = {
						religion = ROOT
						has_province_modifier = estate_dhimmi_guaranteed_autonomy_mod
						AND = {
							religion = orthodox
							has_province_modifier = estate_burghers_guaranteed_autonomy_mod
						}
						AND = {
							culture = ROOT
							ROOT = { is_free_or_tributary_trigger = yes }
						}
					}
				}
				remove_province_modifier = local_religious_difficulties
			}
			every_owned_province = {
				limit = {
					has_province_modifier = estate_nobles_evho_placated_religious_seperatists_mod
					evho_province_is_in_ireland = yes
					OR = {
						religion = ROOT
						has_province_modifier = estate_dhimmi_guaranteed_autonomy_mod
						AND = {
							religion = orthodox
							has_province_modifier = estate_burghers_guaranteed_autonomy_mod
						}
						AND = {
							culture = ROOT
							ROOT = { is_free_or_tributary_trigger = yes }
						}
					}
				}
				remove_province_modifier = estate_nobles_evho_placated_religious_seperatists_mod
			}
			every_owned_province = {
				limit = {
					has_province_modifier = estate_church_evho_right_to_inquisition_mod
					evho_province_is_in_ireland = yes
					OR = {
						religion = ROOT
						has_province_modifier = estate_dhimmi_guaranteed_autonomy_mod
						AND = {
							religion = orthodox
							has_province_modifier = estate_burghers_guaranteed_autonomy_mod
						}
						AND = {
							culture = ROOT
							ROOT = { is_free_or_tributary_trigger = yes }
						}
					}
				}
				remove_province_modifier = estate_church_evho_right_to_inquisition_mod
			}
		}
	}

	option = {
		name = "ev_religion.1.a"
	}
