# Franco-Ottoman alliance
country_event = {
	id = historical_friends_mod.1
	title = "historical_friends_mod.1.t"
	desc = "historical_friends_mod.1.d"
	picture = DIPLOMACY_eventPicture	
	hidden = yes
	fire_only_once = yes
	trigger = {
		is_year = 1530
		BYZ = { exists = no }
		LAE = { exists = no }
		FRA = { exists = yes is_free_or_tributary_trigger = yes }
		TUR = { exists = yes is_free_or_tributary_trigger = yes }
		NOT = {
			OR = {
				is_year = 1680
				AND = {
					is_year = 1580
					NOT = {
						any_province = {
							region = anatolia_region
							OWNER = { 
								is_rival = FRA
								is_rival = TUR
							}
						}
					}
					NOT = {
						any_province = {
							region = france_region
							OWNER = { 
								is_rival = FRA
								is_rival = TUR
							}
						}
					}
				}
			}	
		}
	}
	mean_time_to_happen = {
		days = 1
	}
	 option = {	
		name = "historical_friends_mod.1.o"
		ai_chance = {
			factor = 3
		}
	}	
	 immediate = {
		FRA = { 
			remove_historical_rival = TUR 
			add_historical_friend = TUR 
			if = {
				limit = {
					FRA = { ai = yes }
					TUR = { ai = yes }
				}
				create_alliance = TUR
			}
			add_trust = {
			    who = TUR	
			    value = 400
			    mutual = yes
			}
			add_opinion = {
			    who = TUR
			    modifier = new_friends
			}
			set_ai_attitude = {
				who = TUR
				attitude = attitude_friendly
			}
		}
		TUR = { 
			remove_historical_rival = FRA 
			add_historical_friend = FRA
			add_trust = {
			    who = FRA	
			    value = 400
			    mutual = yes
			}
			add_opinion = {
			    who = FRA
			    modifier = new_friends
			}
			set_ai_attitude = {
				who = FRA
				attitude = attitude_friendly
			}
		}
	}
}
