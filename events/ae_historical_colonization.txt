# Castile/Spain - annex Aztecs
country_event = {
	id = ae_historical_colonization.100
	title = ae_historical_colonization.100.t
	desc = ae_historical_colonization.100.d
	picture = EXPLORERS_eventPicture 
	
	fire_only_once = yes
	
	trigger = { 
		is_year = 1519
		OR = {
			AND = {
				tag = CAS
				NOT = { exists = SPA }
			}
			tag = SPA
		}
		colonial_mexico = {
			ROOT = { has_discovered = PREV }
		}
		colonial_the_carribean = {
			owned_by = ROOT
		}
		NOT = { ae_gamerule_historical_expansion_off = yes }
		OR = {
			ae_gamerule_historical_expansion_on = yes 
			all_countries_including_self = {
				if = {
					limit = { 
						OR = {
							tag = PREV
							tag = CAS
							tag = SPA
							tag = AZT
							any_owned_province = { 
								OR = {
									region = mexico_region
									region = central_america_region
								}
							}
							any_core_province = { 
								OR = {
									region = mexico_region
									region = central_america_region
								}
							}
						}
					}
					full_ai = yes
				}
			}
		}
		
	}
	
	mean_time_to_happen = {
		months = 12
	}	
	
	option = { 
		name = ae_historical_colonization.100.a 
		every_country = {
			limit = {
				capital_scope = {
					region = mexico_region
				}
				any_owned_province = {
					OR = {
						area = mexico_area
						area = eastern_mexico_area
						area = puebla_area
						area = huasteca_area 
					}
				}
			}
			ROOT = { inherit = PREV }
		} 
		AZT = {
			if = {
				limit = { exists = yes }
				ROOT = { inherit = PREV }
			}
		}
	}
}

	id = ae_historical_colonization.102
	title = ae_historical_colonization.102.t
	desc = ae_historical_colonization.102.d
	picture = iberian_COLONIZATION_eventPicture 
	
	fire_only_once = yes
	
	trigger = { 
		is_year = 1499
		OR = {
			AND = {
				tag = CAS
				NOT = { exists = SPA }
			}
			tag = SPA
		}
		colonial_the_carribean = {
			any_province = {
				ROOT = { has_discovered = PREV }
			}
		}
		OR = {
			cuba_area = { NOT = { owned_by = ROOT } }
			east_cuba_area = { NOT = { owned_by = ROOT } }
		}
		NOT = { ae_gamerule_historical_expansion_off = yes }
		OR = {
			ae_gamerule_historical_expansion_on = yes 
			all_countries_including_self = {
				if = {
					limit = { 
						OR = {
							tag = PREV
							tag = CAS
							tag = SPA
							any_owned_province = { 
								colonial_region = colonial_the_carribean
							}
							any_core_province = { 
								colonial_region = colonial_the_carribean
							}
						}
					}
					full_ai = yes
				}
			}
		}
	}
	
	mean_time_to_happen = {
		months = 12
	}	
	
	option = { 
		name = ae_historical_colonization.200.a 
		cuba_area = { 
			cede_province = ROOT
			add_core = ROOT
		}
		east_cuba_area = { 
			cede_province = ROOT
			add_core = ROOT
		}
	}
}
