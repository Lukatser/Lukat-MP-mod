	id = ev_americas.5
	title = ev_americas.5.t
	desc = ev_americas.5.d
	picture = RELIGION_eventPicture
	
	trigger = {
		tag = USA
		is_at_war = no
		is_free_or_tributary_trigger = yes
		religion_group = christian
		NOT = { has_country_flag = america_is_reformed }
		OR = {
			NOT = { religion = reformed }
			NOT = {
				num_of_owned_provinces_with = {
					value = 100
					religion = reformed
				}
			}
		}
	}
	
	major = yes
	fire_only_once = yes
	mean_time_to_happen = {
		years = 1
	}

	option = {
		name = ev_americas.5.a
		set_country_flag = america_is_reformed
		hidden_effect = {
			capital_scope = { change_religion = reformed }
			if = {
				limit = { NOT = { religion = reformed } }
				every_owned_province = {
					limit = { 
						has_owner_religion = yes 
						OR = {
							continent = north_america
							continent = south_america
							continent = new_world
						}
					}
					change_religion = reformed
				}
			}
			else = {
				every_owned_province = {
					limit = { 
						has_owner_accepted_culture = yes 
						OR = {
							continent = north_america
							continent = south_america
							continent = new_world
						}
						OR = {
							has_port = yes
							development = 10
						}
					}
					change_religion = reformed
				}
			}
		}
		change_religion = reformed
	}
}

country_event = {
	id = ev_americas.6
	title = ev_americas.6.t
	desc = ev_americas.6.d
	picture = CONGRESS_eventPicture
	
	major = yes
	trigger = {
		tag = USA
		is_year = 1820
		is_great_power = yes
		has_country_flag = USA_slavery_legal
		any_owned_province = { area = upper_louisiana_area }
		is_free_or_tributary_trigger = yes
		OR = {
			government = republic
			has_parliament = yes
		}
		any_owned_province = {
			has_province_modifier = usa_slavery
			OR = {
				all_canadian_colonies = yes
				region = california_region
				region = cascadia_region
				region = great_lakes_region
				region = northeast_america_region
				region = great_plains_region
			}
		}
	}
	
	fire_only_once = yes
	mean_time_to_happen = {
		months = 12
	}

	option = {
		name = ev_americas.6.a # Prohibit slavery!
		every_owned_province = {
			limit = {
				NOT = { has_province_modifier = northwest_ordinance_slavery_prohibited }
				OR = {
					all_canadian_colonies = yes
					region = california_region
					region = cascadia_region
					region = great_lakes_region
					region = northeast_america_region
					region = great_plains_region
				}
			}
			add_province_modifier = {
				name = north_and_territories_slavery_prohibited
				duration = -1
			}
		}
		hidden_effect = {
			set_country_flag = USA_slavery_banned_in_north
			every_owned_province = {
				limit = {
					has_province_modifier = usa_slavery
					OR = {
						all_canadian_colonies = yes
						region = california_region
						region = cascadia_region
						region = great_lakes_region
						region = northeast_america_region
						region = great_plains_region
					}
				}
				remove_province_modifier = usa_slavery
			}
		}
	}
	option = {
		name = ev_americas.6.b # Let them decide for themselves
		ai_chance = { factor = 0 }
		add_stability = -6
		every_owned_province = {
			limit = {
				OR = {
					all_canadian_colonies = yes
					region = california_region
					region = cascadia_region
					region = great_lakes_region
					region = northeast_america_region
					region = great_plains_region
				}
			}
			add_province_modifier = {
				name = add_unrest_5_modifier
				duration = 1825
			}
		}
	}
}

# I wish I was in Dixie
country_event = {
	id = ev_americas.7
	title = ev_americas.7.t
	desc = ev_americas.7.d
	picture = COLONIZATION_eventPicture
	
	major = yes
	trigger = {
		is_year = 1800
		is_free_or_tributary_trigger = yes
		NOT = { has_country_flag = has_dixie_culture }
		any_owned_province = {
			dixie_can_spawn_here = yes
			culture = american
		}
	}
	
	mean_time_to_happen = {
		years = 15
		modifier = {
			factor = 0.5
			is_year = 1815
		}
		modifier = {
			factor = 0.1
			is_year = 1830
		}
		modifier = {
			factor = 0.01
			has_global_flag = dixie_exists
		}
	}

	option = {
		name = ev_americas.7.a
		set_country_flag = has_dixie_culture
		if = {
			limit = { NOT = { has_global_flag = dixie_exists } }
			set_global_flag = dixie_exists
		}
		if = {
			limit = { capital_scope = { dixie_can_spawn_here = yes } }
			change_primary_culture = confederate
			set_ruler_culture = confederate
			if = {
				limit = {
					has_heir = yes
				}
				set_heir_culture = confederate
			}
			if = {
				limit = {
					has_consort = yes
					OR = {
						consort_culture = english
						consort_culture = scottish
						consort_culture = welsh
						consort_culture = irish
						consort_culture = highland_scottish
						consort_culture = cornish
						consort_culture = canadian
						consort_culture = american
					}
				}
				set_consort_culture = confederate
			}
		}
		else = {
			add_accepted_culture = confederate
		}
		hidden_effect = {
			every_owned_province = {
				limit = { 
					dixie_can_spawn_here = yes
					culture = american
				}
				change_culture = confederate
			}
		}
	}
}
