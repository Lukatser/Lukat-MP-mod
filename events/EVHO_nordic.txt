#War for Independence

country_event = {
	id = EVHO_nordic_events.2
	title = EVHO_nordic_events.2.title
	desc = EVHO_nordic_events.2.desc
	picture = CIVIL_WAR_eventPicture
	
	is_triggered_only = yes
	
	major = yes
	
	option = {
		name = EVHO_nordic_events.2.a
		every_country = {
			limit = { is_supporting_independence_of = ROOT }
			create_alliance = ROOT
		}
		ROOT = { grant_independence = yes }
		DAN = {
			declare_war_with_cb = {
				who = ROOT
				casus_belli = cb_restore_personal_union
			}
		}
		set_country_flag = already_got_a_vasa
		remove_country_modifier = swe_any_vasa_leads_the_rebellion
		if = {
			limit = {
				has_leader = "Johan Vasa"
			}
			define_leader_to_ruler = {
				name = "Johan Vasa"
				adm = 6
				dip = 4
				mil = 4
				claim = 50
				fixed = yes
			}
		}
		else_if = {
			limit = {
				has_leader = "Erik Vasa"
			}
			define_leader_to_ruler = {
				name = "Erik Vasa"
				adm = 6
				dip = 4
				mil = 4
				claim = 50
				fixed = yes
			}
		}
		else_if = {
			limit = {
				has_leader = "Gustav Vasa"
			}
			define_leader_to_ruler = {
				name = "Gustav Vasa"
				adm = 6
				dip = 4
				mil = 4
				claim = 50
				fixed = yes
			}
		}
		else_if = {
			limit = {
				NOT = { is_year = 1490 }
			}
			define_ruler = {
				name = "Johan I"
				dynasty = "Vasa"
				age = 25
				adm = 6
				dip = 4
				mil = 4
			}
		}
		else_if = {
			limit = {
				NOT = { is_year = 1520 }
			}
			define_ruler = {
				name = "Erik I"
				dynasty = "Vasa"
				age = 25
				adm = 6
				dip = 4
				mil = 4
			}
		}
		else = {
			define_ruler = {
				name = "Gustav I"
				dynasty = "Vasa"
				age = 25
				adm = 6
				dip = 4
				mil = 4
			}
			hidden_effect = {
				add_ruler_personality = lawgiver_personality
				add_ruler_personality = industrious_personality
			}
		}
		add_government_reform = autocracy_reform
		add_ruler_modifier = {
			name = swe_rise_of_a_new_dynasty
			duration = -1
		}
		remove_heir = {}
		hidden_effect = {
			capital_scope = {
				build_to_forcelimit = {
					infantry = 0.9
					cavalry = 0.9
					artillery = 0.3
				}
			}
			HSA = {
				country_event = {
					id = EVHO_nordic_events.5
					days = 180
				}
			}
			if = {
				limit = { SWE = { ai = yes } }
				every_province = {
					limit = {
						country_or_non_sovereign_subject_holds = DAN 
						OR = {
							region = baltic_region
							region = russia_region
							area = finland_area
							area = bothnia_area
							area = laponia_area
							area = norrland_area
							area = svealand_area
							area = ostra_svealand_area
							area = vastra_gotaland_area
							area = gotaland_area
							province_id = 25
						}
					}
					add_core = SWE
					cede_province = SWE
				}
			}
			if = {
				limit = {
					SWE = { ai = yes }
					DAN = { ai = yes }
				}
				every_province = {
					limit = {
						OR = {
							region = baltic_region
							region = russia_region
							area = finland_area
							area = bothnia_area
							area = laponia_area
							area = norrland_area
							area = svealand_area
							area = ostra_svealand_area
							area = vastra_gotaland_area
							area = gotaland_area
							province_id = 25
						}
					}
					remove_core = DAN
					remove_claim = DAN
				}
			}
		}
	}
	after = {
		clr_country_flag = swe_support_from_Johan_vasa
		clr_country_flag = swe_support_from_Erik_vasa
		clr_country_flag = swe_support_from_Gustav_vasa
	}
}

		set_country_flag = SWE_forced_into_parliamentarianism
		ai_chance = {
			factor = 100
		}
	}
}
