country_event = {
	id = super_colonies.7
	title = "super_colonies.RHtit1"
	desc = "super_colonies.RHdesc1"
	picture = DEBATE_REPUBLICAN_eventPicture
	
	trigger = {
		NOT = {
			any_country = {
				war_with = ROOT
				OR = {
					is_great_power = yes
					capital_scope = { continent = north_america }
				}
			}
		}
		is_year = 1750
		is_colonial_nation = no
		#Canada super colony
		AND = {
			any_subject_country = {
				#ai = yes
				is_colonial_nation = yes
				is_colonial_nation_of = ROOT
				capital_scope = {
					colonial_region = colonial_canada
				}
				NOT = { has_country_flag = formed_super_colony }
				#NOT = { liberty_desire = 90 }
			}
			4890 = { country_or_non_sovereign_subject_holds = ROOT }
			993 = { country_or_non_sovereign_subject_holds = ROOT }
			994 = { country_or_non_sovereign_subject_holds = ROOT }
			973 = { country_or_non_sovereign_subject_holds = ROOT }
			num_of_provinces_owned_or_owned_by_non_sovereign_subjects_with = {
				colonial_region = colonial_hudson_bay
				value = 10
			}
			colonial_alberta = {
				type = all
				country_or_non_sovereign_subject_holds = ROOT
				is_city = yes
			}
		}
	}
	
	mean_time_to_happen = {
		months = 2
	}
	
	option = {
		name = "super_colonies.RHans1"	
		#Canada super colony
		every_country = {
			limit = {
				is_colonial_nation = yes
				is_colonial_nation_of = ROOT
				capital_scope = {
					OR = {
						colonial_region = colonial_acadia
						colonial_region = colonial_hudson_bay
						colonial_region = colonial_alberta
						colonial_region = colonial_columbia
					}
				}
			}
			every_country = {
				limit = {
					is_colonial_nation = yes
					is_colonial_nation_of = ROOT
					capital_scope = {
						colonial_region = colonial_canada
					}
				}
				set_country_flag = formed_super_colony
				set_country_flag = formed_canada_colony
				add_prestige = 25
				inherit = PREV
				if = {
					limit = { ROOT = { culture_group = british } }
					4890 = {
						add_base_tax = 3 
						add_base_production = 2
						add_base_manpower = 2
						change_culture = canadian
						change_religion = ROOT
					}
					set_capital = 4890 #Capital now Ottowa
					change_primary_culture = canadian
					if = {
						limit = { NOT = { has_global_flag = canadian_culture_exists } }
						set_global_flag = canadian_culture_exists
					}
				}
				else_if = {
					limit = { ROOT = { culture_group = french } }
					994 = {
						add_base_tax = 3 
						add_base_production = 2
						add_base_manpower = 2
						change_culture = creole
						change_religion = ROOT
					}
					set_capital = 994 #Capital now Quebec City
					change_primary_culture = creole
				}
				hidden_effect = {
					if = {
						limit = { has_custom_ideas = no }
						country_event = { id = ideagroups.1 } #Swap Ideas
					}
					colonial_canada = {
						add_permanent_claim = PREV
					}
					colonial_acadia = {
						add_permanent_claim = PREV
					}
					colonial_hudson_bay = {
						add_permanent_claim = PREV
					}
					colonial_alberta = {
						add_permanent_claim = PREV
					}
					colonial_columbia = {
						add_permanent_claim = PREV
					}
					2005 = {
						add_permanent_claim = PREV
					}
					2598 = {
						add_permanent_claim = PREV
					}
					
					add_country_modifier = {
						name = "centralization_modifier"
						duration = 7300
