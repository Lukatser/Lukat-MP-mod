# Inhospitable Interior - AFRICA for NON-AFRICANS
country_event = {
	id = ae_restricted_expansion.100
	title = ae_restricted_expansion.100.t
	desc = ae_restricted_expansion.100.d
	picture = IMPALED_SOLDIERS_eventPicture 
	
	trigger = { 
		NOT = { ae_gamerule_restricted_expansion_off = yes }
		OR = {
			ae_gamerule_restricted_expansion_on = yes 
			full_ai = yes
		}
		capital_scope = { 
			NOT = {	continent = africa }
			NOT = { superregion = near_east_superregion }
		}
		any_owned_province = {
			NOT = { has_province_flag = ae_exempt_restr_exp }
			continent = africa
			has_port = no
		}
		NOT = {
			OR = {
				is_year = 1800
				FROM = { dip_tech = 32 }
			}
		}
	}
	
	
	is_triggered_only = yes
	
	option = { 
		name = ae_restricted_expansion.100.a  
		every_owned_province = {
			limit = {
				NOT = { has_province_flag = ae_exempt_restr_exp }
				continent = africa
				has_port = no
			}
			
			cede_province = previous_owner
			every_country = { 
				limit = {
					OR = {
						tag = ROOT
						is_subject_of = ROOT
					}
					PREV = {
						OR = {
							is_core = PREV
							is_claim = PREV
						}
					}
				}
				PREV = { 
					remove_core = PREV 
					remove_claim = PREV 
				}
			}
			owner = {
				if = {
					limit = { NOT = { truce_with = ROOT } }
					add_truce_with = ROOT
				}
			}
		} 
	}
}
