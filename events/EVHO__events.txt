country_event = {
	id = EVHO_misc_events.1
	title = EVHO_misc_events.1.t
	desc = EVHO_misc_events.1.d
	picture = BATTLE_eventPicture
	
	is_triggered_only = yes
	
	option = {
		name = EVHO_misc_events.1.a
		FROM = {
			if = {
				limit = { FROM = { is_subject = no } }
				create_alliance = ROOT
			}
			ROOT = { join_all_offensive_wars_of = FROM }
			ROOT = { join_all_defensive_wars_of = FROM }
			set_ai_attitude = {
				who = ROOT
				attitude = attitude_friendly
			}
			ROOT = {
				set_ai_attitude = {
					who = FROM
					attitude = attitude_friendly
				}
			}
		}
		ai_chance = {
			factor = 1
			modifier = {
				factor = 0
				OR = {
					AND = {
						is_at_war = yes
						NOT = { war_score = -25 }
					}
					AND = {
						is_at_war = yes
						is_great_power = no
						NOT = { war_score = -1 }
					}
					AND = {
						is_great_power = yes
						any_country = {
							war_with = ROOT
							is_great_power = yes
							NOT = { tag = ROOT }
						}
					}
					NOT = { has_opinion = { who = FROM value = 0 } }
					any_country = {
						war_with = FROM
						alliance_with = ROOT
					}
				}
			}
		}
	}
	option = {
		name = EVHO_misc_events.1.b
		add_prestige = -50
		ai_chance = {
			factor = 1
			modifier = {
				factor = 0
				OR = {
					FROM = { is_colonial_nation_of = ROOT }
					AND = {
						has_opinion = { who = FROM value = 0 }
						NOT = {
							any_country = {
								war_with = FROM
								alliance_with = ROOT
							}
						}
						OR = {
							is_at_war = no
							AND = {
								is_great_power = no
								war_score = -1
							}
							AND = {
								is_great_power = yes
								NOT = { 
									any_country = {
										war_with = ROOT
										is_great_power = yes
										NOT = { tag = ROOT }
									}
								}
							}
						}
					}
				}
			}
		}
	}
}

		any_country = {
			government = republic
			NOT = {
				OR = {
					has_country_flag = republic_name
					has_country_flag = proclaimed_an_empire
				}
			}
			OR = {
				tag = TUR
				tag = USA
				tag = AFU
				tag = UNK
			}
		}
	}
	
	mean_time_to_happen = {
		months = 1
	}

	option = {
		name = EVHO_misc_events.4.a

		ai_chance = { factor = 1 }
		every_country = {
			limit = {
				government = republic
				NOT = {
					OR = {
						has_country_flag = republic_name
						has_country_flag = proclaimed_an_empire
					}
				}
				OR = {
					tag = TUR

					tag = UNK
				}
			}

				limit = { tag = TUR }

				limit = { tag = USA }
