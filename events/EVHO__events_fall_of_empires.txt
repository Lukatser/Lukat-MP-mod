country_event = {
	id = EVHO_events_fall_of_empires.100
	title = EVHO_events_fall_of_empires.100.t
	desc = EVHO_events_fall_of_empires.100.d
	picture = BAD_WITH_MONARCH_eventPicture
	
	trigger = {
		tag = MNG
		is_great_power = yes
		in_golden_age = no
		any_neighbor_country = {
			is_nomad = yes
			capital_scope = {
				OR = {
					superregion = tartary_superregion
					superregion = far_east_superregion
				}
			}
		}
		OR = {
			AND = {
				has_dlc = "Common Sense"
				government_rank = 3
			}
			AND = {
				NOT = { has_dlc = "Common Sense" }
				total_development = 1000
			}
		}
		NOT = { has_country_modifier = MNG_unguarded_frontier }
		NOT = { has_country_flag = MNG_unguarded_frontier_flag }
		
	}

	mean_time_to_happen = {
		months = 100
		modifier = {
			factor = 0.75
			has_had_golden_age = yes
		}
		modifier = {
			factor = 0.5
			total_development = 1500
		}
		modifier = {
			factor = 0.25
			total_development = 2000
		}
	}

	
	
	option = {
		name = EVHO_events_fall_of_empires.100.a 
		set_country_flag = MNG_unguarded_frontier_flag
		add_country_modifier = {
			name = MNG_unguarded_frontier
			duration = -1  
		}
	}
}
