railroad = {
	cost = 2000
	time = 6
	
	build_trigger = {
		NOT = { has_terrain = glacier }
		NOT = { has_terrain = impassable_mountains }
		NOT = { has_terrain = jungle }
		NOT = { has_terrain = desert }
		NOT = { has_terrain = coastal_desert }
	}
	
	modifier = {
		local_development_cost = -0.1
		province_trade_power_modifier = 0.1
	}
	on_built = {
		if = {
			limit = {
				OR = {
					development = 20
					any_neighbor_province = {
						has_building = railroad
					}
				}
			}
			add_province_modifier = {
				name = evho_province_railroad_modifier
				duration = -1
			}
		}
	}
	on_destroyed = {
		if = {
			limit = { has_province_modifier = evho_province_railroad_modifier }
			remove_province_modifier = evho_province_railroad_modifier
		}
	}

	ai_will_do = {
		factor = 1
		modifier = {
			factor = 0
			owner = {
				NOT = { has_institution = industrialization }
			}
		}
		modifier = {
			factor = 1.1
			army_size_percentage = 0.5
		}
		modifier = {
			factor = 2
			owner = {
				has_institution = revolution_inst
			}
		}
	}
}
