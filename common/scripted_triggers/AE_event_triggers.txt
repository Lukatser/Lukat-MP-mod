# PARTITION OF LIVONIA - actors and province definitions

is_liv_partition_muscovy = {
	if = {
		limit = { RUS = { is_neighbor_of = ROOT } }
		tag = RUS
	}
	else_if = {
		limit = { MOS = { is_neighbor_of = ROOT } }
		tag = MOS
	}
	else = {
		NOT = {
			any_country = {
				NOT = { tag = PREV }
				AND = {
					culture_group = east_slavic
					NOT = {
						primary_culture = ruthenian
						primary_culture = byelorussian
					}
				}
				army_size = PREV
			}
		}
	}
}
