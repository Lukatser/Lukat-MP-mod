become_historic_friends = {
	category = favors
	require_acceptance = no

	is_visible = {
	}
	is_allowed = {
	FROM = {
	NOT = {
		historical_friend_with = ROOT
				}
			}
	FROM = {
	NOT = {
		historical_rival_with = ROOT
				}
			}
	FROM = {
	owes_favors = {
         who = ROOT 
         value = 100
			}
		}
	}
	on_accept = {
	ROOT = {
add_favors = {
   who = FROM
   amount = -100
   }
}
				ROOT = {
					add_historical_friend = FROM
				}
				FROM = {
					add_historical_friend = ROOT
		}
	}
	ai_will_do = {}
}

end_historic_rivalry = {
	category = favors
	require_acceptance = no

	is_visible = {
	}
	is_allowed = {
	FROM = {
		historical_rival_with = ROOT
		}
	FROM = {
	owes_favors = {
         who = ROOT 
         value = 100
			}
		}
	FROM = {
			has_opinion = {
			who = ROOT
			value = 150
			}
		}
	}
	on_accept = {
	ROOT = {
add_favors = {
   who = FROM
   amount = -100
   }
}
				ROOT = {
					remove_historical_rival = FROM
				}
				FROM = {
					remove_historical_rival = ROOT
		}
	}
	ai_will_do = {}
}
