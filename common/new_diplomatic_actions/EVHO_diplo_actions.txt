		has_government_attribute = has_eyalets
		FROM = {
			is_subject_of = ROOT
			is_subject_of_type = core_eyalet
		}
	}
	is_allowed = {
		if = {
			limit = {
				FROM = { liberty_desire = 50 }
			}
			custom_trigger_tooltip = {
				tooltip = FROM_IS_REBELLIOUS_TT
				FROM = { liberty_desire = 50 }
			}
		}
	}
	on_accept = {
		if = {
			limit = { 
				ai = yes 
				FROM = { ai = no }
			}
			FROM = { switch_tag = ROOT }
		}
		inherit = FROM
	}
	on_decline = {
		declare_war_with_cb = {
		    who = FROM
		    casus_belli = cb_crush_rebellion
		}
		hidden_effect = {
			every_country = {
				limit = {
					same_continent = FROM
					OR = {

					}
				}

			}
		}
	}

	ai_will_do = { # this is a trigger
		if = {
			limit = {
				NOT = {
					FROM = {
						OR = {
							tag = WAL
							tag = TRA
							tag = MOL
						}
					}
				}
			}
			always = yes	#There is never a reason for the Ottomans to not want to integrate their eyalets
		}
		else = {
			always = no
		}
	}

	ai_acceptance = {
        add_entry = {
            name = DIPLO_OPPINION
			export_to_variable = {
				variable_name = ai_value
				value = opinion
				who = FROM
				with = ROOT
			}
			multiply_variable = {
				which = ai_value 
				value = 0.1
			}
        }
		add_entry = {
			name = AI_DIPLOMATIC_REPUTATION
			export_to_variable = {
				variable_name = ai_value
				value = trigger_value:diplomatic_reputation
			}
			#1 rep = 2
			multiply_variable = {
				which = ai_value
				value = 2
			}
		}
		add_entry = {
			name = AI_TOTAL_DEVELOPMENT
			export_to_variable = {
				variable_name = ai_value
				value = total_development
				who = FROM
			}

				which = ai_value
