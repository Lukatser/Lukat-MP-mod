#When adding effects directly in on_actions always consider if it would be best to do so through a scripted effect. on_actions with a lot of script in them are harder to get an overview of.

# country
on_startup = {
	#### Dev click counter ###

	every_country = {
		if = {
			limit = {
				check_variable = {
					which = dev_adm_clicks					
					value = 1
				}
			}
		}
		else = {
			set_variable = {
				which = dev_adm_clicks
				value = 0

# province
on_adm_development = {
	on_development_effect = { type = adm }
	owner = {
		country_event = { id = funny.1 }
	}
	#ARUMBA LINES
    hidden_effect = {
		owner = {
			change_variable = {
				which = dev_adm_clicks
				value = 1
			}
			change_variable = {
				which = dev_sum_clicks
				value = 1
			}
		}
	}
}
