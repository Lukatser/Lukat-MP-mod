hre_shadow_kingdom_give_up_italy_effect = {
	add_prestige = -10
	add_imperial_influence = -20
	custom_tooltip = shadow_kingdom_emperor_gives_up_tt
	set_global_flag = italians_leave_hre
	hidden_effect = {
		every_country = {
			limit = {
				is_shadow_kingdom_eligible_trigger = yes
				OR = {
					is_subject = no
					AND = {
						is_subject = yes
						overlord = { # overlord chooses for their vassals if eligible to leave themselves
							NOT = {
								is_part_of_hre = yes
								capital_scope = { region = italy_region }
							}
						}
					}
				}
			}
			country_event = { id = shadow_kingdom.4 }
		}
	}
	every_province = {
		limit = {
			is_part_of_hre = yes
			region = italy_region
			owner = {
				is_part_of_hre = no
			}
		}
		set_in_empire = no
	}
}

hre_shadow_kingdom_leave_hre_effect = {
	set_in_empire = no
	set_country_flag = permanently_left_hre_flag
	emperor = {
		add_opinion = {
			who = PREV
			modifier = hre_country_left
		}
	}
	if = {
		limit = {
			any_subject_country = {
				capital_scope = {
					region = italy_region
				}
				is_part_of_hre = yes
			}
		}
		every_subject_country = {
			limit = {
				capital_scope = {
					region = italy_region
				}
				is_part_of_hre = yes
			}
			set_in_empire = no
		}
	}
}

hree_put_imperial_ban_on_this = {
	clr_country_flag = emperor_used_imperial_ban
	set_country_flag = emperor_used_imperial_ban
	set_country_flag = hree_imperial_ban_country_flag
	set_ruler_flag = hree_imperial_ban
	if = {
		limit = {
			has_regency = yes
			has_heir = yes
		}
		set_heir_flag = hree_imperial_ban
	}
	emperor = {
		add_opinion = {
			who = PREV
			modifier = hree_imperial_ban_emperor_opinion
		}
	}
	every_elector = {
		limit = {
			is_emperor = no
			NOT = { tag = PREV }
		}
		add_opinion = {
			who = PREV
			modifier = hree_imperial_ban_elector_opinion
		}
	}
	every_country = {
		limit = {
			is_part_of_hre = yes
			is_elector = no
			is_emperor = no
			NOT = { tag = PREV }
		}
		add_opinion = {
			who = PREV
			modifier = hree_imperial_ban_member_opinion
		}
	}
	every_country = {
		limit = {
			NOT = { tag = PREV }
			is_subject = no
			is_part_of_hre = yes
			is_revolutionary = no
			NOT = { has_ruler_flag = hree_imperial_ban }
			NOT = { has_heir_flag = hree_imperial_ban }
			is_neighbor_of = PREV
		}
		add_casus_belli = {
			target = PREV
			type = cb_hree_imperial_ban
		}
	}
	country_event = { id = besseres_hrr_events.34 }
}

hree_add_imperial_arch_office = {
	if = {
		limit = {
			NOT = { has_saved_global_event_target = hre_$name$ }
			NOT = { hree_has_any_imperial_archoffice = yes }
		}
		add_country_modifier = {
			name = $name$
			duration = -1
		}
		hidden_effect = {
			save_global_event_target_as = hre_$name$
			add_opinion = {
				who = emperor
				modifier = hree_imperial_archoffice_granted_opinion
			}
			country_event = { id = besseres_hrr_events.39 }
		}
	}
	hree_update_imperial_archoffices = yes
}

					}
				}
			}
			else_if = {
				limit = {
					OR = {
						has_country_modifier = erzbannertraeger
						any_country = {
							has_country_modifier = erzbannertraeger
						}
					}
				}
				every_country = {
					limit = {
						has_country_modifier = erzbannertraeger
					}
					remove_country_modifier = erzbannertraeger
				}
			}
		}
	}
}
